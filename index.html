        <audio

            id="audio"

            controls

            autoplay

        ></audio>

    <script>


        const streamUrl =

            "https://pub-3ce7b3bcc5e844aa9da0768b47c5b711.r2.dev/live/stream.m3u8";


        const audio = document.getElementById("audio");


        const status = document.getElementById("status");


        if (Hls.isSupported()) {


            const hls = new Hls({


                // =========================

                // IMPORTANT BUFFER SETTINGS

                // =========================


                liveSyncDuration: 15,


                liveMaxLatencyDuration: 25,


                maxBufferLength: 30,


                maxMaxBufferLength: 60,


                backBufferLength: 90,


                lowLatencyMode: false,


            });


            hls.loadSource(streamUrl);


            hls.attachMedia(audio);


            // =========================

            // Stream Ready

            // =========================


            hls.on(Hls.Events.MANIFEST_PARSED, function () {


                status.innerText =

                    "Connected to live commentary";


                audio.play();


            });


            // =========================

            // Error Handling

            // =========================


            hls.on(Hls.Events.ERROR, function (event, data) {


                console.log("HLS Error:", data);


                if (data.fatal) {


                    switch(data.type) {


                        case Hls.ErrorTypes.NETWORK_ERROR:


                            status.innerText =

                                "Network issue... reconnecting";


                            hls.startLoad();


                            break;


                        case Hls.ErrorTypes.MEDIA_ERROR:


                            status.innerText =

                                "Recovering audio stream";


                            hls.recoverMediaError();


                            break;


                        default:


                            status.innerText =

                                "Stream disconnected";


                            hls.destroy();


                            break;

                    }

                }

            });


        }


        else if (audio.canPlayType('application/vnd.apple.mpegurl')) {


            // Safari native HLS support


            audio.src = streamUrl;


            audio.addEventListener('loadedmetadata', function () {


                audio.play();


            });

        }
